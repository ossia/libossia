437inline ossia::value init_value(ossia::val_type type)
438{
439 switch(type)
440 {
441 case val_type::IMPULSE:
442 return ossia::impulse{};
443 case val_type::BOOL:
444 return bool{};
445 case val_type::INT:
446 return int32_t{};
447 case val_type::FLOAT:
448 return float{};
449 case val_type::STRING:
450 return value{std::string{}}; // value needed for explicit ctor
451 case val_type::LIST:
452 return value{std::vector<ossia::value>{}};
453 case val_type::MAP:
454 return value{value_map_type{}};
455 case val_type::VEC2F:
456 return vec2f{};
457 case val_type::VEC3F:
458 return vec3f{};
459 case val_type::VEC4F:
460 return vec4f{};
461 case val_type::NONE:
462 default:
463 break;
464 }
465
466 ossia_do_throw(invalid_value_type_error, "init_value: Invalid type");
467 return {};
